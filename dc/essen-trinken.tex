\begin{example}
\begin{tabularx}{\linewidth}{XV{3}X}
													& Dort ist ein Tisch frei. \\
	\ro												& Ich brauche jetzt \ex{einen Kaffee}. \\
	\ro	\multirow{-2}{*}{Was möchten Sie trinken?}	& Ich möchte bittet \ex{einen Kaffee}. \\
	Mit \ex{Milch} und \ex{Zucker}?					& Mit viel \ex{Mich} und ohne \ex{Zucker}. \\
	\ro	Ist das alles?								& Nein, ich nehme noch \fillhere . \\
	Also \ex{einen Kaffee} mit Milch und \fillhere	& Ja, bitte. \\
\end{tabularx}
\end{example}
