\begin{syntax}{Satzbau}{}
\begin{itemize}
	\item Aussagesatz:
		\begin{mdframed}
			Personalpronomen + \verbhere + \fillhere
		\end{mdframed}
	\item Ja-Nein-Fragen
		\begin{mdframed}
			\verbhere + Personalpronomen + \fillhere ?
		\end{mdframed}
	\item W-Fragen
		\begin{mdframed}
			\begin{tabular}{l}
				Woher \\
				Wo \\
				Wie \\
				Welche
			\end{tabular} + \verbhere + Personalpronomen ?
		\end{mdframed}
\end{itemize}
\end{syntax}
